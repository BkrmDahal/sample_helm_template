```bash
bash start.sh
```

2. Set env from secret
```yaml

- name: AWS_DEFAULT_REGION
    valueFrom:
    secretKeyRef:
        name: aws
        key: region

# pass args
        args:
          - --database
          - redis
          - --redis
          - "redis://redis:6379/4"
          - --port
          - "8080"
```
